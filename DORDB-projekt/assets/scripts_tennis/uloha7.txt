-- tabulka Zapas, Ucastnik_zapasu
CREATE OR REPLACE PROCEDURE vytvor_zapas (id_hrac_domaci IN NUMBER, id_hrac_hoste IN NUMBER, datum IN DATE) AS
pocet_domaci NUMBER;
pocet_hoste NUMBER;
pocet_celkem NUMBER;
idstadion NUMBER;
BEGIN
  SELECT divaci INTO pocet_domaci FROM Tym WHERE ID_Tym = id_hrac_domaci;
  SELECT divaci INTO pocet_hoste FROM Tym WHERE ID_Tym = id_hrac_hoste;
  
  pocet_celkem := pocet_domaci + pocet_hoste;
  
  SELECT * INTO idstadion FROM(SELECT ID_stadion FROM Stadion WHERE kapacita > pocet_celkem AND ID_stadion NOT IN (SELECT ID_stadion FROM Zapas WHERE datum = cas) ORDER BY kapacita ASC) WHERE rownum = 1;
  
  INSERT INTO Zapas VALUES (seq_Zapas.nextval, datum, idstadion);
  INSERT INTO Ucastnik_zapasu VALUES (seq_Ucastnik_zapasu.nextval, id_hrac_domaci, seq_Zapas.currval);
  INSERT INTO Ucastnik_zapasu VALUES (seq_Ucastnik_zapasu.nextval, id_hrac_hoste, seq_Zapas.currval);
END;
